77 {

80 
81  const char *expr = iexpr;
82  unsigned int n = strlen(expr);
83  char *rexpr = static_cast<char*>(malloc(3*n));
84  char *rpnexpr = rexpr;
85 
86  unsigned int token_len;
87  char token[256];
88 
89  while (*expr)
90  {
91  // skip white space
92  if (isspace(*expr))
93  {
94  while(*expr && isspace(*expr)) ++expr;
95  }
96  // recurse into grouped expression
97  else if ((token_len = tokenizer_t::get_open_group(expr, token)))
98  {
99  token[token_len-1] = '\0';
100 
101  char *tmp = infix_to_postfix<tokenizer_t>(token+1, variables);
102  unsigned int tmp_len = strlen(tmp);
103 
104  memcpy(rexpr, tmp, tmp_len);
105  rexpr += tmp_len;
106 
107  expr += token_len;
108 
109  free(tmp);
110  }
111  // pass constants through
112  else if ((token_len = tokenizer_t::get_constant_name(expr, token)))
113  {
114  memcpy(rexpr, token, token_len);
115  rexpr += token_len;
116  *rexpr++ = ' ';
117 
118  expr += token_len;
119  }
120  // pass variable names through, save the variable name
121  else if ((token_len = tokenizer_t::get_variable_name(expr, token)))
122  {
123  if (variables)
124  variables->insert(token);
125 
126  memcpy(rexpr, token, token_len);
127  rexpr += token_len;
128  *rexpr++ = ' ';
129 
130  expr += token_len;
131  }
132  // push operator names onto the stack
133  else if ((token_len = tokenizer_t::get_ternary_operator_name(expr, token))
134  || (token_len = tokenizer_t::get_binary_operator_name(expr, token))
135  || (token_len = tokenizer_t::get_unary_operator_name(expr, token)))
136  {
137  // apply precedence rules. operators of higher precedence
138  // are popped and applied
139  unsigned int p1 = tokenizer_t::get_operator_precedence(token);
140  while (operator_stack.size() &&
141  (tokenizer_t::get_operator_precedence(operator_stack.back()) >= p1))
142  {
143  char *op_name = operator_stack.back();
144  unsigned int op_len = strlen(op_name);
145 
146  operator_stack.pop_back();
147 
148  memcpy(rexpr, op_name, op_len);
149  rexpr += op_len;
150  *rexpr++ = ' ';
151 
152  free(op_name);
153  }
154 
155  // push the new operator
156  operator_stack.push_back(strdup(token));
157  expr += token_len;
158  }
159  // every other input indicates an error
160  else
161  {
162  TECA_SYNTAX_ERROR(iexpr, expr-iexpr)
163  return nullptr;
164  }
165  }
166 
167  // catch unmatched open group
168  if (group_position.size())
169  {
170  TECA_SYNTAX_ERROR(iexpr, group_position.back())
171  return nullptr;
172  }
173 
174  // apply the remaining operators
175  while (operator_stack.size())
176  {
177  char *op_name = operator_stack.back();
178  unsigned int op_len = strlen(op_name);
179 
180  operator_stack.pop_back();
181 
182  memcpy(rexpr, op_name, op_len);
183  rexpr += op_len;
184  *rexpr++ = ' ';
185 
186  free(op_name);
187  }
188

190  *rexpr = '\0';
191  return rpnexpr;
